│ 1.3 Flujo de Datos Principal                                                                                                                                                                           │
│                                                                                                                                                                                                        │
│ Flujo de Reserva:                                                                                                                                                                                      │
│ 1. Usuario busca → API → Supabase (RLS) → Resultados                                                                                                                                                   │
│ 2. Usuario selecciona → Detalle vehículo + Calendario                                                                                                                                                  │
│ 3. Usuario reserva → Server Action → Validaciones                                                                                                                                                      │
│ 4. Crear reserva (pending) → Mercado Pago Preference                                                                                                                                                   │
│ 5. Usuario paga → MP Webhook → Actualizar reserva                                                                                                                                                      │
│ 6. Notificación email (host + driver) → Reserva confirmada                                                                                                                                             │
│ 7. Post-alquiler → Liberar pago → Reviews bidireccionales                                                                                                                                              │
│                                                                                                                                                                                                        │

│ 4. Diseño Técnico: Pagos Escrow                                                                                                                                                                        │
│                                                                                                                                                                                                        │
│ 4.1 Flujo de Escrow con Mercado Pago                                                                                                                                                                   │
│                                                                                                                                                                                                        │
│ ┌─────────────────────────────────────────────────────────────────────────────┐                                                                                                                        │
│ │                         FLUJO DE PAGO ESCROW                                │                                                                                                                        │
│ └─────────────────────────────────────────────────────────────────────────────┘                                                                                                                        │
│                                                                                                                                                                                                        │
│ 1. RESERVA CREADA                                                                                                                                                                                      │
│    Driver → Crear Booking (status: PENDING)                                                                                                                                                            │
│           → Crear Payment (status: PENDING)                                                                                                                                                            │
│           → Generar MP Preference                                                                                                                                                                      │
│                                                                                                                                                                                                        │
│ 2. USUARIO PAGA                                                                                                                                                                                        │
│    Driver → MP Checkout                                                                                                                                                                                │
│           → MP Webhook: payment.approved                                                                                                                                                               │
│           → Payment.status = PROCESSING                                                                                                                                                                │
│           → Payment.escrowStatus = HELD                                                                                                                                                                │
│           → Booking.status = CONFIRMED                                                                                                                                                                 │
│           → Emails: Host + Driver                                                                                                                                                                      │
│                                                                                                                                                                                                        │
│ 3. ALQUILER COMPLETADO                                                                                                                                                                                 │
│    Sistema → 24h post return                                                                                                                                                                           │
│           → Si sin incidentes:                                                                                                                                                                         │

│ 4.2 Modelo de Comisión                                                                                                                                                                                 │
│                                                                                                                                                                                                        │
│ // Configuración de comisiones                                                                                                                                                                         │
│ interface CommissionConfig {                                                                                                                                                                           │
│   platformFeePercent: number;     // Default: 15%                                                                                                                                                      │
│   minimumFee: number;             // En centavos: 500 ($5)                                                                                                                                             │
│   maximumFee: number;             // En centavos: 10000 ($100)                                                                                                                                         │
│   depositPercent: number;         // Default: 10% del valor vehículo                                                                                                                                   │
│   weekendMarkup: number;          // Default: 20%                                                                                                                                                      │
│ }                                                                                                                                                                                                      │
│                                                                                                                                                                                                        │
│ // Cálculo de reserva                                                                                                                                                                                  │
│ function calculateBookingAmount(                                                                                                                                                                       │
│   vehicle: Vehicle,                                                                                                                                                                                    │
│   startDate: Date,                                                                                                                                                                                     │
│   endDate: Date,                                                                                                                                                                                       │
│   config: CommissionConfig                                                                                                                                                                             │
│ ): BookingAmount {                                                                                                                                                                                     │
│   const days = differenceInDays(endDate, startDate);                                                                                                                                                   │
│   const isWeekend = checkWeekend(startDate, endDate);                                                                                                                                                  │
│                                                                                                                                                                                                        │
│   // Precio base                                                                                                                                                                                       │
│   const dailyPrice = isWeekend                                                                                                                                                                         │
│     ? (vehicle.weekendPriceDay || vehicle.basePriceDay * 1.2)                                                                                                                                          │
│     : vehicle.basePriceDay;                                                                                                                                                                            │
│                                                                                                                                                                                                        │
│   const baseAmount = dailyPrice * days;                                                                                                                                                                │
│                                                                                                                                                                                                        │
│   // Comisión plataforma                                                                                                                                                                               │
│   let platformFee = baseAmount * (config.platformFeePercent / 100);                                                                                                                                    │
│   platformFee = Math.max(config.minimumFee, Math.min(platformFee, config.maximumFee));                                                                                                                 │
│                                                                                                                                                                                                        │
│   // Depósito                                                                                                                                                                                          │
│   const depositAmount = vehicle.estimatedValue * (config.depositPercent / 100);                                                                                                                        │
│                                                                                                                                                                                                        │
│   return {                                                                                                                                                                                             │
│     baseAmount,                                                                                                                                                                                        │
│     platformFee,                                                                                                                                                                                       │
│     hostPayout: baseAmount - platformFee,                                                                                                                                                              │
│     depositAmount,                                                                                                                                                                                     │
│     totalAmount: baseAmount + depositAmount,                                                                                                                                                           │
│   };                                                                                                                                                                                                   │
│ }                                                                                                                                                                                                      │
│                                                                                                                                                                                                        │
